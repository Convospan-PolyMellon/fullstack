import { NextRequest, NextResponse } from "next/server";
import { prisma } from "@/lib/db"; // stubbed, but works
import { upsertDocuments } from "@/ai/vectorStore";

export async function POST(req: NextRequest) {
  try {
    const body = await req.json();
    const { name, title, url } = body;

    if (!name) {
      return NextResponse.json(
        { error: "Missing profile name" },
        { status: 400 }
      );
    }

    // save into vector store for long-term RAG learning
    await upsertDocuments([
      {
        id: `linkedin-${name.replace(/\s+/g, "_").toLowerCase()}`,
        content: `LinkedIn Profile:\nName: ${name}\nTitle: ${title}\nURL: ${url}`,
      },
    ]);

    return NextResponse.json({
      status: "ok",
      message: "Profile synced",
      profile: body,
    });
  } catch (err: any) {
    return NextResponse.json(
      { error: err.message || "unknown error" },
      { status: 500 }
    );
  }
}
